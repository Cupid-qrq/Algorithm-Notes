## 题1（100分）

### 思路
- 你原程序的核心判断是：
  - `fact`: 两个序列是否完全相同。
  - `judge`: 两个序列映射值（按下标取值后异或）是否相同。
  - 若 `fact == judge` 输出 `correct`，否则 `wrong`。
- 优化点：
  - 读入时直接算每组异或值，避免重复遍历。
  - 使用变长二维数组存序列，避免固定大矩阵和尾部0干扰。

### 代码

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m;
  cin >> n >> m;

  vector<long long> val(n + 1);
  for (int i = 1; i <= n; i++) cin >> val[i];

  vector<vector<int>> S(m + 1), T(m + 1);
  vector<long long> xs(m + 1, 0), xt(m + 1, 0);

  for (int i = 1; i <= m; i++) {
    int len;
    cin >> len;
    S[i].resize(len);
    long long x = 0;
    for (int j = 0; j < len; j++) {
      cin >> S[i][j];
      x ^= val[S[i][j]];
    }
    xs[i] = x;
  }

  for (int i = 1; i <= m; i++) {
    int len;
    cin >> len;
    T[i].resize(len);
    long long x = 0;
    for (int j = 0; j < len; j++) {
      cin >> T[i][j];
      x ^= val[T[i][j]];
    }
    xt[i] = x;
  }

  for (int i = 1; i <= m; i++) {
    bool fact = (S[i] == T[i]);
    bool judge = (xs[i] == xt[i]);
    cout << (fact == judge ? "correct" : "wrong") << '\n';
  }

  return 0;
}
```

## 题2（100分）

### 思路
- 关键观察：`g(x, k)` 只依赖 `x` 的低9位（3段各3bit），输出也在 `[0, 511]`。
- 所以整条变换链等价于一个 512 状态映射 `F`。
- 对每个终态 `y`，预处理最小初态 `x`（非负）满足 `F(x) = y`，即可 O(1) 回答查询。

### 代码

```cpp
#include <bits/stdc++.h>
using namespace std;

static inline int f3(int x, int k) {
  return ((x * x + k * k) & 7) ^ (k & 7);
}

static inline int trans(int x, int k) {
  int c = x & 7;
  int b = (x >> 3) & 7;
  int a = (x >> 6) & 7;

  int na = a ^ f3(c, k);
  int nc = c ^ f3(b, k);

  return na | (nc << 3) | (b << 6);
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m;
  cin >> n >> m;

  vector<int> k(m + 1), ask(n + 1);
  for (int i = 1; i <= m; i++) cin >> k[i];
  for (int i = 1; i <= n; i++) cin >> ask[i];

  const int S = 512;
  vector<int> F(S);
  for (int x = 0; x < S; x++) {
    int cur = x;
    for (int i = 1; i <= m; i++) cur = trans(cur, k[i]);
    F[x] = cur;
  }

  // best[y] = 最小 x 使 F[x] = y
  vector<int> best(S, -1);
  for (int x = 0; x < S; x++) {
    int y = F[x];
    if (best[y] == -1 || x < best[y]) best[y] = x;
  }

  for (int i = 1; i <= n; i++) {
    int y = ask[i];
    int ans = (0 <= y && y < S) ? best[y] : -1;
    cout << ans << (i == n ? '\n' : ' ');
  }

  return 0;
}
```

## 题3（从75到100：坐标修正 + 合并全局旋转）

### 思路
- 你当前75分版本超时核心：每次遇到类型1都旋转整张 `z*z` 矩阵，复杂度高。
- 满分做法：
  - 维护 `global_rot`（整图已累计顺时针旋转次数 mod 4）。
  - 不真的旋整图，只在读写单元格时做坐标映射（逻辑坐标 -> 物理坐标）。
  - 子矩阵旋转/翻转都在逻辑坐标下执行，内部通过映射访问数组。
- 这样可把大量整图操作降掉。

### 代码

```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 405;
char g[N][N];
int z, k, t;

struct Op {
  int type;
  int a, b, c, d, e;
};

vector<Op> ops;
int global_rot = 0;  // 累计整图顺时针旋转次数 mod 4

static inline pair<int, int> map_pos(int x, int y) {
  if (global_rot == 0) return {x, y};
  if (global_rot == 1) return {z - y + 1, x};
  if (global_rot == 2) return {z - x + 1, z - y + 1};
  return {y, z - x + 1};
}

static inline char get_cell(int x, int y) {
  auto [px, py] = map_pos(x, y);
  return g[px][py];
}

static inline void set_cell(int x, int y, char v) {
  auto [px, py] = map_pos(x, y);
  g[px][py] = v;
}

// 逻辑坐标下，把子方阵 [u..u+l-1][v..v+l-1] 顺时针旋转 d 次(90度/次)
void whirl_sub(int u, int v, int l, int d) {
  d %= 4;
  if (d == 0) return;

  static char tmp[N][N];
  for (int i = 0; i < l; i++) {
    for (int j = 0; j < l; j++) {
      tmp[i][j] = get_cell(u + i, v + j);
    }
  }

  if (d == 1) {
    for (int i = 0; i < l; i++) {
      for (int j = 0; j < l; j++) {
        set_cell(u + j, v + (l - 1 - i), tmp[i][j]);
      }
    }
  } else if (d == 2) {
    for (int i = 0; i < l; i++) {
      for (int j = 0; j < l; j++) {
        set_cell(u + (l - 1 - i), v + (l - 1 - j), tmp[i][j]);
      }
    }
  } else {
    for (int i = 0; i < l; i++) {
      for (int j = 0; j < l; j++) {
        set_cell(u + (l - 1 - j), v + i, tmp[i][j]);
      }
    }
  }
}

// 逻辑坐标下翻转
// o==1: 上下翻转; o==2: 左右翻转
void change_rect(int u, int d, int l, int r, int o) {
  int h = d - u + 1;
  int w = r - l + 1;
  if (o == 1) {
    for (int i = 0; i < h / 2; i++) {
      for (int j = l; j <= r; j++) {
        char x = get_cell(u + i, j);
        char y = get_cell(d - i, j);
        set_cell(u + i, j, y);
        set_cell(d - i, j, x);
      }
    }
  } else {
    for (int j = 0; j < w / 2; j++) {
      for (int i = u; i <= d; i++) {
        char x = get_cell(i, l + j);
        char y = get_cell(i, r - j);
        set_cell(i, l + j, y);
        set_cell(i, r - j, x);
      }
    }
  }
}

void print_ans(int n, int m) {
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) cout << get_cell(i, j);
    cout << '\n';
  }
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  cin >> z;
  for (int i = 1; i <= z; i++) {
    for (int j = 1; j <= z; j++) cin >> g[i][j];
  }

  cin >> k >> t;
  ops.resize(t);
  for (int i = 0; i < t; i++) {
    int op, u, v, l, d, r;
    cin >> op >> u >> v >> l >> d >> r;
    ops[i] = {op, u, v, l, d, r};
  }

  // 逆序还原
  for (int i = t - 1; i >= 0; i--) {
    const auto &cur = ops[i];
    if (cur.type == 1) {
      int u = cur.a, v = cur.b, l = cur.c, d = cur.d, r = cur.e;

      // 原程序语义: 先对整图做 r 次顺时针(逆过程里按同方向累计到坐标系)
      global_rot = (global_rot + ((r % 4) + 4) % 4) % 4;

      // 子方阵旋转做逆操作
      whirl_sub(u, v, l, (4 - (d / 90) % 4) % 4);
    } else {
      int u = cur.a, d = cur.b, l = cur.c, r = cur.d, o = cur.e;
      change_rect(u, d, l, r, o);
    }
  }

  int n = -1, m = -1;
  for (int i = 1; i <= z; i++) {
    if (get_cell(i, 1) == '?') {
      n = i - 1;
      break;
    }
  }
  if (n == -1) n = z;

  for (int j = 1; j <= z + 1; j++) {
    if (get_cell(1, j) == '?') {
      m = j - 1;
      break;
    }
  }
  if (m == -1) m = z;

  cout << n << ' ' << m << '\n';
  print_ans(n, m);
  return 0;
}
```
